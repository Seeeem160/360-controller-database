<script>
  const params = new URLSearchParams(window.location.search);
  const controllerId = params.get("id");

  const content = document.getElementById("content");

  if (!controllerId) {
    content.innerHTML = `
      <div class="not-found">
        <h1>No controller selected</h1>
        <p>Please open this page from the controller list.</p>
      </div>
    `;
    throw new Error("Missing controller ID");
  }

  fetch("data/controllers.json")
    .then(res => res.json())
    .then(data => {
      const controller = data.find(c => c.id === controllerId);

      if (!controller) {
        content.innerHTML = `
          <div class="not-found">
            <h1>Controller not found</h1>
            <p>The requested controller does not exist in the database.</p>
          </div>
        `;
        return;
      }

      const rarityClass =
        controller.rarity === "Ultra Rare" ? "ultra" :
        controller.rarity === "Rare" ? "rare" : "";

      content.innerHTML = `
        <h1>${controller.name}</h1>

        <div class="meta">
          <span class="badge ${rarityClass}">${controller.rarity}</span>
          <span class="badge">${controller.category}</span>
          <span class="badge">${controller.release_year}</span>
          <span class="badge">${controller.connection}</span>
        </div>

        <section>
          <h2>General Information</h2>
          <p><strong>Colors:</strong> ${controller.colors.join(", ")}</p>
          <p><strong>Regions:</strong> ${controller.regions.join(", ")}</p>
          <p><strong>Shell finish:</strong> ${controller.shell_finish}</p>
          ${controller.part_numbers.length ? `<p><strong>Part numbers:</strong> ${controller.part_numbers.join(", ")}</p>` : ""}
          ${controller.serial_notes ? `<p><strong>Serial notes:</strong> ${controller.serial_notes}</p>` : ""}
        </section>

        ${controller.authentication.length ? `
        <section>
          <h2>Authentication Checklist</h2>
          <ul>
            ${controller.authentication.map(item => `<li>${item}</li>`).join("")}
          </ul>
        </section>` : ""}

        ${controller.collector_notes ? `
        <section>
          <h2>Collector Notes</h2>
          <p>${controller.collector_notes}</p>
        </section>` : ""}

        ${controller.known_fakes ? `
        <div class="warning">
          <strong>⚠ Warning:</strong> This controller is commonly counterfeited.
        </div>` : ""}

        ${controller.images.length ? `
        <section>
          <h2>Images</h2>
          <div class="images">
            ${controller.images.map(src => `<img src="${src}" alt="">`).join("")}
          </div>
        </section>` : ""}
      `;
    })
    .catch(err => {
      content.innerHTML = `
        <div class="not-found">
          <h1>Error loading controller</h1>
          <p>Check your JSON file path and syntax.</p>
        </div>
      `;
      console.error(err);
    });
</script>
